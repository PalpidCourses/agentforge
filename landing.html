Deploy in 60 seconds.

Three commands. Real infrastructure.

bash 
# 1. Clone and run
git clone https://github.com/D0NMEGA/agentforge.git
cd agentforge
pip install -r requirements.txt
uvicorn main:app --host 0.0.0.0 --port 8000

# 2. Register your agent
curl -X POST http://localhost:8000/v1/register \
  -H "Content-Type: application/json" \
  -d '{"name": "my-bot"}'

# 3. Store persistent state
curl -X POST http://localhost:8000/v1/memory \
  -H "X-API-Key: af_YOUR_KEY" \
  -H "Content-Type: application/json" \
  -d '{"key":"state","value":"{\"mood\":\"bullish\"}"}'
python — SDK example 
from agentforge import AgentForge

# Register a new agent (one line)
result = AgentForge.register(name="my-bot")
print(result["api_key"])  # Save this — cannot be recovered

# Create a client
af = AgentForge(api_key="af_your_key_here")

# Persistent memory with TTL
af.memory_set("portfolio", '{"BTC": 1.5}', ttl_seconds=86400)

# Bot-to-bot messaging
af.send_message("agent_abc123", {"signal": "buy"})
messages = af.inbox()

# Task queue with priorities
af.queue_submit({"task": "scrape"}, priority=8)

# Cron scheduling
af.schedule_create("*/5 * * * *", {"task": "check_prices"})

# Shared memory for cross-agent data
af.shared_set("market_data", "BTC_price", "98500")

# Agent directory
af.directory_update(description="Price tracker", capabilities=["alerts"])

# Webhooks
af.webhook_create("https://my-server.com/hook", ["message.received"])
