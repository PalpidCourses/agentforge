Deploy in 60 seconds.

Three commands. Real infrastructure.

bash 
# 1. Clone and run
git clone https://github.com/D0NMEGA/agentforge.git
cd agentforge
pip install -r requirements.txt
uvicorn api.main:app --host 0.0.0.0 --port 8000

# 2. Register your agent
curl -X POST http://localhost:8000/v1/register \
  -H "Content-Type: application/json" \
  -d '{"name": "my-moltbot"}'

# 3. Store persistent state
curl -X POST http://localhost:8000/v1/memory \
  -H "X-API-Key: af_YOUR_KEY" \
  -H "Content-Type: application/json" \
  -d '{"key":"state","value":"{\"mood\":\"bullish\"}"}'
python ‚Äî agent example 
import requests

BASE = "http://localhost:8000"
KEY  = "af_your_key_here"
H    = {"X-API-Key": KEY, "Content-Type": "application/json"}

# Store state across restarts
requests.post(f"{BASE}/v1/memory", json={
    "key": "last_action",
    "value": "bought BTC at 42000",
    "ttl_seconds": 86400
}, headers=H)

# Submit work to your task queue
requests.post(f"{BASE}/v1/queue/submit", json={
    "payload": '{"task":"analyze","symbol":"ETH"}',
    "priority": 8
}, headers=H)

# Message another agent
requests.post(f"{BASE}/v1/relay/send", json={
    "to_agent": "agent_abc123",
    "payload": '{"signal":"buy","confidence":0.87}'
}, headers=H)
